--------------------------------------------------------------------------------

Escribe una función llamada `ft_scanf` que reproduzca el comportamiento del
scanf estándar de C, con las siguientes restricciones:

- Solo gestionará las siguientes conversiones: s, d y c
- No tienes que gestionar las opciones *, m ni '
- No tienes que gestionar el ancho máximo de campo
- No tienes que gestionar los modificadores de tipo (h, hh, l, etc.)
- No tienes que gestionar las conversiones con la sintaxis %n$

Tu función debe declararse de la siguiente forma:

int ft_scanf(const char *, ...);

En este directorio encontrarás un archivo con parte del código que necesitarás,
solo tienes que completarlo.

Para comprobar tu implementación, compara los resultados de tu ft_scanf con los
del scanf real de la librería estándar.

Pista: puede que necesites consultar el manual de scanf (man 3 scanf).

int	match_conv(FILE *f, const char **format, va_list ap)
{
	switch (**format)
	{
		case 'c':
			return scan_char(f, ap);
		case 'd':
			match_space(f);
			return scan_int(f, ap);
		case 's':
			match_space(f);
			return scan_string(f, ap);
		case EOF:
			return -1;
		default:
			return -1;
	}
}

int ft_vfscanf(FILE *f, const char *format, va_list ap)
{
	int nconv = 0;

	int c = fgetc(f);
	if (c == EOF)
		return EOF;
	ungetc(c, f);

	while (*format)
	{
		if (*format == '%')
		{
			format++;
			if (match_conv(f, &format, ap) != 1)
				break;
			else
				nconv++;
		}
		else if (isspace(*format))
		{
			if (match_space(f) == -1)
				break;
		}
		else if (match_char(f, *format) != 1)
			break;
		format++;
	}

	if (ferror(f))
		return EOF;
	return nconv;
}


int ft_scanf(const char *format, ...)
